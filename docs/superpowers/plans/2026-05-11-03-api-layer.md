# API Layer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the AWS Lambda API layer — experiment CRUD, trial status, results streaming via WebSocket, and SQS job dispatch — that connects the frontend to the optimization engine.

**Architecture:** API Gateway (REST + WebSocket) routes to Lambda handlers. Experiment CRUD persists to DynamoDB. On experiment start, a message is sent to SQS which triggers an ECS Fargate task. The engine publishes progress events to a DynamoDB stream; a Lambda streams those to connected WebSocket clients.

**Tech Stack:** Python 3.12, boto3, AWS Lambda, API Gateway, DynamoDB, SQS, pytest, moto (AWS mocking)

**Prerequisite:** Plan 01 (shared gene schema) must be complete.

from __future__ import annotations
import json
import uuid
from datetime import datetime, timezone


def _now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def _api_response(status_code: int, body: dict) -> dict:
    return {
        "statusCode": status_code,
        "headers": {
            "Content-Type": "application/json",
            "Access-Control-Allow-Origin": "*",
        },
        "body": json.dumps(body),
    }


def ok(body: dict) -> dict:
    return _api_response(200, body)


def created(body: dict) -> dict:
    return _api_response(201, body)


def not_found(message: str = "Not found") -> dict:
    return _api_response(404, {"error": message})


def bad_request(message: str) -> dict:
    return _api_response(400, {"error": message})


def server_error(message: str = "Internal server error") -> dict:
    return _api_response(500, {"error": message})


def new_experiment_id() -> str:
    return f"exp_{uuid.uuid4().hex[:12]}"
```

from __future__ import annotations
import json
import os
import boto3
from backend.shared.experiment import ExperimentConfig, ObjectiveWeights, EvaluatorConfig
from backend.api.models import ok, created, not_found, bad_request, server_error, new_experiment_id, _now_iso

_TABLE = os.environ.get("EXPERIMENTS_TABLE", "autoaw-experiments")


def _table():
    return boto3.resource("dynamodb").Table(_TABLE)


def create(event: dict, context) -> dict:
    try:
        body = json.loads(event.get("body") or "{}")
        config = ExperimentConfig(
            name=body["name"],
            task_description=body["task_description"],
            dataset_id=body["dataset_id"],
            evaluators=[EvaluatorConfig.from_dict(e) for e in body["evaluators"]],
            objective_weights=ObjectiveWeights.from_dict(body["objective_weights"]),
            population_size=body.get("population_size", 20),
            budget_max_trials=body.get("budget_max_trials"),
            budget_max_usd=body.get("budget_max_usd"),
            convergence_patience=body.get("convergence_patience", 10),
            concurrency=body.get("concurrency", 5),
        )
    except (KeyError, ValueError) as e:
        return bad_request(str(e))

    exp_id = new_experiment_id()
    now = _now_iso()
    _table().put_item(Item={
        "pk": exp_id,
        "sk": "config",
        "experiment_id": exp_id,
        "name": config.name,
        "status": "draft",
        "created_at": now,
        "updated_at": now,
        "config_json": json.dumps(config.to_dict()),
    })
    return created({"experiment_id": exp_id, "status": "draft", "created_at": now})


def get(event: dict, context) -> dict:
    exp_id = event["pathParameters"]["experiment_id"]
    resp = _table().get_item(Key={"pk": exp_id, "sk": "config"})
    item = resp.get("Item")
    if not item:
        return not_found(f"Experiment {exp_id} not found")
    config = json.loads(item["config_json"])
    return ok({
        "experiment_id": item["experiment_id"],
        "name": item["name"],
        "status": item["status"],
        "created_at": item["created_at"],
        "updated_at": item["updated_at"],
        "config": config,
        "best_gene_id": item.get("best_gene_id"),
        "best_fitness": item.get("best_fitness"),
    })


def list_all(event: dict, context) -> dict:
    result = _table().scan(FilterExpression="sk = :sk", ExpressionAttributeValues={":sk": "config"})
    experiments = [
        {
            "experiment_id": item["experiment_id"],
            "name": item["name"],
            "status": item["status"],
            "created_at": item["created_at"],
        }
        for item in result.get("Items", [])
    ]
    experiments.sort(key=lambda x: x["created_at"], reverse=True)
    return ok({"experiments": experiments})


def delete(event: dict, context) -> dict:
    exp_id = event["pathParameters"]["experiment_id"]
    _table().delete_item(Key={"pk": exp_id, "sk": "config"})
    return ok({"deleted": exp_id})
```

from __future__ import annotations
import json
import os
import boto3
from backend.api.models import ok, not_found, server_error, _now_iso

_EXPERIMENTS_TABLE = os.environ.get("EXPERIMENTS_TABLE", "autoaw-experiments")
_JOBS_QUEUE_URL = os.environ.get("JOBS_QUEUE_URL", "")


def _experiments_table():
    return boto3.resource("dynamodb").Table(_EXPERIMENTS_TABLE)


def _update_status(exp_id: str, status: str) -> None:
    _experiments_table().update_item(
        Key={"pk": exp_id, "sk": "config"},
        UpdateExpression="SET #s = :s, updated_at = :t",
        ExpressionAttributeNames={"#s": "status"},
        ExpressionAttributeValues={":s": status, ":t": _now_iso()},
    )


def start(event: dict, context) -> dict:
    exp_id = event["pathParameters"]["experiment_id"]
    table = _experiments_table()
    resp = table.get_item(Key={"pk": exp_id, "sk": "config"})
    if not resp.get("Item"):
        return not_found(f"Experiment {exp_id} not found")

    # Send job to SQS; the Fargate task consumer picks it up
    sqs = boto3.client("sqs")
    sqs.send_message(
        QueueUrl=_JOBS_QUEUE_URL,
        MessageBody=json.dumps({"experiment_id": exp_id, "action": "start"}),
    )
    _update_status(exp_id, "running")
    return ok({"experiment_id": exp_id, "status": "running"})


def stop(event: dict, context) -> dict:
    exp_id = event["pathParameters"]["experiment_id"]
    sqs = boto3.client("sqs")
    sqs.send_message(
        QueueUrl=_JOBS_QUEUE_URL,
        MessageBody=json.dumps({"experiment_id": exp_id, "action": "stop"}),
    )
    _update_status(exp_id, "stopped")
    return ok({"experiment_id": exp_id, "status": "stopped"})
```

from __future__ import annotations
import json
import os
import boto3
from boto3.dynamodb.conditions import Key
from backend.api.models import ok

_TRIALS_TABLE = os.environ.get("TRIALS_TABLE", "autoaw-trials")


def _table():
    return boto3.resource("dynamodb").Table(_TRIALS_TABLE)


def list_for_experiment(event: dict, context) -> dict:
    exp_id = event["pathParameters"]["experiment_id"]
    resp = _table().query(
        KeyConditionExpression=Key("pk").eq(exp_id) & Key("sk").begins_with("trial#"),
        ScanIndexForward=False,
        Limit=200,
    )
    items = resp.get("Items", [])
    trials = [
        {
            "gene_id": item["gene_id"],
            "generation": int(item["generation"]),
            "fitness": float(item["fitness"]),
            "quality": float(item["quality"]),
            "cost_usd": float(item["cost_usd"]),
            "latency_ms": int(item["latency_ms"]),
        }
        for item in items
    ]
    return ok({"trials": trials, "experiment_id": exp_id})


def get_trial(event: dict, context) -> dict:
    exp_id = event["pathParameters"]["experiment_id"]
    gene_id = event["pathParameters"]["gene_id"]
    resp = _table().query(
        KeyConditionExpression=Key("pk").eq(exp_id) & Key("sk").begins_with(f"trial#{gene_id}#"),
        Limit=1,
    )
    items = resp.get("Items", [])
    if not items:
        from backend.api.models import not_found
        return not_found(f"Trial {gene_id} not found in experiment {exp_id}")
    item = items[0]
    return ok({
        "gene_id": item["gene_id"],
        "generation": int(item["generation"]),
        "fitness": float(item["fitness"]),
        "gene": json.loads(item["gene_json"]),
    })
```
